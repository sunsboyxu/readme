const cos = new COS({
	SecretId: '',
	SecretKey: ''
})

cos.putObject({
	Bucket: '存储桶名称',
	Region: '地区简写',
	Key: '文件路径+文件名',
	Body: 'file 文件对象',
	onProgress: function(progressData) {
		console.log(progressData)
	} 
}, function(err, data) { // 上传后的回调
	// 处理上传失败
	if(err) {
		console.log(err)
	}
	// 处理成功
	console.log(data)
})
